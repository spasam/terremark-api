209         if (organizationId == null) {
210             getOrganizations();
211         }
212 
213         final CreateSshKey createSshKey = new CreateSshKey();
214         createSshKey.setDefault(Boolean.FALSE);
215         createSshKey.setName(keyName);
216 
217         SshKey sshKey = client.getOrganizationHandler().sshKeyCreate(organizationId, createSshKey);
218         assertNotNull("SSH key not created");
219         assertResource(sshKey);
220         assertNotNull("Invalid SSH private key: " + sshKey, sshKey.getPrivateKey());
221         assertNotNull("Invalid SSH fingerprint: " + sshKey, sshKey.getFingerPrint());
222 
223         OutputStream os = new FileOutputStream(System.getProperty("java.io.tmpdir") + "/" + keyName + ".pem");
224         try {
225             os.write(sshKey.getPrivateKey().getBytes());
226         } finally {
227             os.close();
228         }
229 
230         sshKeyName = keyName;
231         sshKeyId = TerremarkFactory.getId(sshKey);
232 
233         return TerremarkFactory.getReference(sshKey);
234     }

315                 try {

317                 } catch (InterruptedException ex) {
318                     // Ignore
319                 }
320 
321                 t = client.getEnvironmentHandler().getTaskByID(taskId);
322                 assertResourceNoName(t);
323 
324                 switch (t.getStatus()) {
325                 case COMPLETE:
326                     return;
327                 case ERROR:
328                     throw new Exception("Task failed with error: " + t);
329                 case QUEUED:
330                 case RUNNING:
331                     // Continue
332                 }
333             }
334         }
